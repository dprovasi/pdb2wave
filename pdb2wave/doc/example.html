mu-opioid receptor, from active structure, chain A

Calculate distances from right/left reference residues to mimick sound localisation

data_5c1m = get_structure_pdb(
  "~/5c1m_a.pdb",
  refL_resno = 229,
  refR_resno = 315)

wave_stereo = gen_tune(
  data_5c1m$seq, 
  data_5c1m$ses, 
  distL = data_5c1m$distL, 
  distR = data_5c1m$distR)

plot(wave_stereo)
play(wave_stereo, player = "afplay")
writeWave(wave_stereo, filename="5c1m.wav")
